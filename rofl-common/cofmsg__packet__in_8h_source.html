231  std::string
232  s_reason() const {
233  std::string ss;
234  switch (get_version()) {
235  case rofl::openflow10::OFP_VERSION: {
236  if (rofl::openflow10::OFPR_NO_MATCH == get_reason())
237  ss.append("no match ");
238  if (rofl::openflow10::OFPR_ACTION == get_reason())
239  ss.append("action ");
240  } break;
241  case rofl::openflow12::OFP_VERSION: {
242  if (rofl::openflow12::OFPR_NO_MATCH == get_reason())
243  ss.append("no match ");
244  if (rofl::openflow12::OFPR_ACTION == get_reason())
245  ss.append("action ");
246  if (rofl::openflow12::OFPR_INVALID_TTL == get_reason())
247  ss.append("invalid ttl ");
248  } break;
249  case rofl::openflow13::OFP_VERSION: {
250  if (rofl::openflow13::OFPR_NO_MATCH == get_reason())
251  ss.append("no match ");
252  if (rofl::openflow13::OFPR_ACTION == get_reason())
253  ss.append("action ");
254  if (rofl::openflow13::OFPR_INVALID_TTL == get_reason())
255  ss.append("invalid ttl ");
256  } break;
257  default: {
258  if (rofl::openflow14::OFPR_TABLE_MISS == get_reason())
259  ss.append("table miss ");
260  if (rofl::openflow14::OFPR_APPLY_ACTION == get_reason())
261  ss.append("apply action ");
262  if (rofl::openflow14::OFPR_INVALID_TTL == get_reason())
263  ss.append("invalid ttl ");
264  if (rofl::openflow14::OFPR_ACTION_SET == get_reason())
265  ss.append("action set ");
266  if (rofl::openflow14::OFPR_GROUP == get_reason())
267  ss.append("group ");
268  if (rofl::openflow14::OFPR_PACKET_OUT == get_reason())
269  ss.append("packet out ");
270  };
271  }
272  return ss;
273  };
